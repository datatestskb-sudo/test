‚úÖ Secure Example
JavaScript 
// GOOD: Secure authentication
import bcrypt from 'bcrypt';
import jwt from 'jsonwebtoken';

// GOOD: Hash passwords
async function hashPassword(password) {
  return await bcrypt.hash(password, 12);
}

// GOOD: Validate inputs
function validateInput(input) {
  const sanitized = input.trim();
  if (sanitized.length < 8) {
    throw new Error('Password too short');
  }
  return sanitized;
}

// GOOD: Use JWT tokens
function generateToken(user) {
  return jwt.sign(
    { userId: user.id, role: user.role },
    process.env.JWT_SECRET,
    { expiresIn: '1h' }
  );
}
‚úÖ Security Benefits:
  • Password hashing
  • Input validation

// SECURE: Input sanitization
function sanitizeAIInput(input) {
  const blocked = ['ignore', 'forget', 'system'];
  if (blocked.some(word => 
    input.toLowerCase().includes(word)
  )) {
    throw new Error('Invalid input');
  }
  return input;
}

// SECURE: Controlled prompts
const response = await openai.chat.completions.create({
  model: "gpt-3.5-turbo",

üì± Use Case 1: E-commerce Platform

‚ùå Attack Scenario

Attacker exploits payment API to modify prices

‚úÖ Prevention

API rate limiting + price validation

üõ°Ô∏è Tools Used

AWS WAF, JWT tokens, HMAC

Implementation Example 
// E-commerce Security Implementation
app.post('/api/payment', authenticateUser, async (req, res) => {
  // 1. Validate input
  const { amount, productId } = req.body;
  if (!amount || amount <= 0) {
    return res.status(400).json({ error: 'Invalid amount' });
  }
  
  // 2. Verify price from database (not client)
  const product = await prisma.product.findUnique({
    where: { id: productId },
    select: { price: true }
  });
  
  // 3. Check if amount matches
  if (amount !== product.price) {
    return res.status(403).json({ error: 'Price mismatch' });
  }
  
  // 4. Process payment securely
  const payment = await processSecurePayment(amount);
  
  // 5. Log for audit
  await logSecurityEvent('payment', req.user.id, amount);
});

üè• Use Case 2: Healthcare Application

‚ùå Attack Scenario

‚úÖ Prevention

Healthcare Security 
// Healthcare Security Implementation
class PatientDataAccess {
  async getPatientData(patientId, userRole, userId) {
    // 1. Check authorization
    if (!(await this.isAuthorized(userRole, patientId, userId))) {
      throw new Error('Unauthorized access attempt');
    }
    
    // 2. Get patient data with encryption
    const patient = await prisma.patient.findUnique({
      where: { id: patientId },
      select: {
        id: true,
        name: true,
        medicalRecords: true // Encrypted at rest
      }
    });
    
    // 3. Log access for audit
    await auditLog('patient_data_access', {
      userId,
      patientId,
      timestamp: new Date(),
      action: 'view'
    });
    
    return patient;
  }
  
  async isAuthorized(role, patientId, userId) {
    // Check role-based permissions
    const permissions = {
      'doctor': ['view_medical_records', 'edit_treatment'],
      'nurse': ['view_basic_info', 'edit_vitals'],
      'patient': ['view_own_data']
    };
    
    return permissions[role]?.includes('view_medical_records');
  }
}

FinTech Security 
// FinTech Security Implementation
class TransactionSecurity {
  async processTransaction(userId, amount, recipient) {
    // 1. Verify MFA
    if (!(await this.verifyMFA(userId))) {
      throw new Error('MFA verification failed');
    }
    
    // 2. Check transaction limits
    const dailyLimit = await this.getDailyLimit(userId);
    const currentSpending = await this.getDailySpending(userId);
    
    if (currentSpending + amount > dailyLimit) {
      throw new Error('Transaction exceeds daily limit');
    }
    
    // 3. Verify recipient
    if (!(await this.verifyRecipient(userId, recipient))) {
      throw new Error('Recipient verification failed');
    }
    
    // 4. Process with fraud detection
    const transaction = await this.createSecureTransaction(userId, amount, recipient);
    
    // 5. Real-time monitoring
    await this.fraudDetection(userId, amount, recipient);
    
    return transaction;
  }
  
  async fraudDetection(userId, amount, recipient) {
    const riskScore = await calculateRiskScore({
      userId,
      amount,
      recipient,
      timeOfDay: new Date().getHours(),
      location: await getUserLocation(userId)
    });
    
    if (riskScore > 0.8) {
      await flagTransaction(userId, 'HIGH_RISK_TRANSACTION');
      throw new Error('Transaction flagged for review');
    }
  }
}
